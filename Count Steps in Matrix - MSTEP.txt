#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n, count=0;
		cin>>n;
		int r[n*n+1], c[n*n+1];
		for(int i=0; i<n; i++){
			for(int j=0; j<n; j++){
				int m;
				cin>>m;
				r[m]=i;
				c[m]=j;
			}
		}
		for(int i=1; i<n*n; i++){
			count += abs(r[i]-r[i+1])+abs(c[i]-c[i+1]);
		}
		cout<<count<<endl;
	}
	return 0;
}
